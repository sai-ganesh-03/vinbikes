CREATE TABLE users (
    id INT AUTO_INCREMENT PRIMARY KEY,
    username VARCHAR(100) UNIQUE,
    password VARCHAR(255)
);

CREATE TABLE bikes (
    id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(100),
    model VARCHAR(100),
    price DECIMAL(10, 2),
    image_url VARCHAR(255),
    description TEXT
);

CREATE TABLE custom_bikes (
    id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    customization_level_1 VARCHAR(255),
    customization_level_2 VARCHAR(255),
    customization_level_3 VARCHAR(255),
    description TEXT,
    image_url VARCHAR(255),
);
